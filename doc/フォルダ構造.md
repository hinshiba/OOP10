## 作成コマンド

```bash
# 1. ソリューション作成
dotnet new sln -n MipsDebugger

# 2. プロジェクト作成
dotnet new classlib -n MipsDebugger.Shared -o src/MipsDebugger.Shared
dotnet new classlib -n MipsDebugger.Compiler -o src/MipsDebugger.Compiler
dotnet new classlib -n MipsDebugger.Runtime -o src/MipsDebugger.Runtime
dotnet new classlib -n MipsDebugger.Debugger -o src/MipsDebugger.Debugger
dotnet new avalonia.app -n MipsDebugger.Presentation -o src/MipsDebugger.Presentation

dotnet new xunit -n MipsDebugger.Compiler.Tests -o tests/MipsDebugger.Compiler.Tests
dotnet new xunit -n MipsDebugger.Runtime.Tests -o tests/MipsDebugger.Runtime.Tests
dotnet new xunit -n MipsDebugger.Debugger.Tests -o tests/MipsDebugger.Debugger.Tests
dotnet new xunit -n MipsDebugger.Integration.Tests -o tests/MipsDebugger.Integration.Tests

# 3. ソリューションに追加
dotnet sln add src/**/*.csproj
dotnet sln add tests/**/*.csproj

# 4. プロジェクト間参照設定
dotnet add src/MipsDebugger.Compiler reference src/MipsDebugger.Shared
dotnet add src/MipsDebugger.Runtime reference src/MipsDebugger.Shared
dotnet add src/MipsDebugger.Debugger reference src/MipsDebugger.Shared
dotnet add src/MipsDebugger.Debugger reference src/MipsDebugger.Runtime
dotnet add src/MipsDebugger.Presentation reference src/MipsDebugger.Shared

# テストプロジェクトの参照
dotnet add tests/MipsDebugger.Compiler.Tests reference src/MipsDebugger.Compiler
dotnet add tests/MipsDebugger.Runtime.Tests reference src/MipsDebugger.Runtime
dotnet add tests/MipsDebugger.Debugger.Tests reference src/MipsDebugger.Debugger

# 5. フォルダ作成
mkdir -p docs/requirements docs/design
```
